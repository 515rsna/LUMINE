import socket sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) # 서버 소켓 생성 # server serverIp = input("서버(내 PC)IP를 입력해 주세요(cmd 창에서 ipconfig 로 확인): ") serverPort = int(input("사용할 Port를 입력해 주세요.: ")) sock.bind((serverIp, serverPort)) # 서버의 IP 와 port no. 데이터 받을 준비 print("listen: ", serverPort , "port") while True: data, addr= sock.recvfrom(65535) # 수신 블럭 크기 print("ip: ", addr[0], "port:", addr[1], "로 부터 송신된 메시지는 아래와 같습니다.") # 클라이언트의 ip 와 port print(data.decode()) sock.sendto(data, addr) if data.decode() == "QUIT" : break
